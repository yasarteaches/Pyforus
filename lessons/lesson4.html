üíª Code Examples

# Defining a function
def greet():
    print("Hello, welcome!")

# Calling the function
greet()

# Function with parameter
def greet_user(name):
    print("Hello, " + name + "!")

greet_user("Mira")

# Function with return value
def add(a, b):
    return a + b

result = add(3, 4)
print(result)
