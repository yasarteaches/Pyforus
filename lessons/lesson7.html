  "name": "Ali",
  "age": 16,
  "grades": [85, 90, 92]
}

print(student["name"])        # Ali
student["age"] = 17            # Update value
student["city"] = "Istanbul"   # Add new key-value pair

for key in student:
    print(key, ":", student[key])
